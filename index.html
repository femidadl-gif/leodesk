Налаштування автосинхронізації

Автозбереження при кожній зміні
Коли увімкнено — дані автоматично записуються в Sheets після кожного додавання або видалення
📋 Як отримати URL скрипту
1 Відкрий script.google.com → натисни "Новий проєкт"
2 Видали весь код і встав код зі скриньки нижче
3 Натисни "Розширення" → "Таблиці Google" → "Нова таблиця"
4 Натисни "Розгорнути" → "Нове розгортання" → Тип: Веб-застосунок → Доступ: Всі → "Розгорнути"
5 Скопіюй URL веб-застосунку і встав вище
Код для скрипту (скопіюй повністю)
function doPost(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName('LexDesk') || ss.insertSheet('LexDesk');
    sheet.clearContents();
    sheet.getRange(1,1).setValue(e.postData.contents);
    return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName('LexDesk');
    const val = sheet ? sheet.getRange(1,1).getValue() : '{}';
    return ContentService.createTextOutput(val || '{}')
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput('{}')
      .setMimeType(ContentService.MimeType.JSON);
  }
}
